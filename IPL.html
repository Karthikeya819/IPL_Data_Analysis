seasons = sorted(set(df['season']))

deliveries_group = df1.groupby('match_id')

    runs=0
    season_group = seasons_group.get_group(season)['id']
    for match in season_group:
        runs+=deliveries_group.get_group(match)['total_runs'].sum()
        runs+=deliveries_group.get_group(match)['extra_runs'].sum()
    print(str(season)+' - '+str(runs))

18) Visualize the Toss decision across seasons¶

In [78]:
seasons = sorted(set(df['season']))
x = range(len(seasons))
seasons_group = df[['toss_decision','season']].groupby('season')
group2_data = [seasons_group.get_group(seasons[i])['toss_decision'].count() for i in x]
group1_data = [(seasons_group.get_group(seasons[i])['toss_decision']=='field').sum() for i in x]
group3_data = [(~(seasons_group.get_group(seasons[i])['toss_decision']=='field')).sum() for i in x]
bar_width = 0.3

plt.bar(x, group1_data, width=bar_width, label='Field', align='center')
plt.bar([i+ bar_width for i in x], group2_data, width=bar_width, label='Total Tosses', align='center')
plt.bar([i + 2 * bar_width for i in x], group3_data, width=bar_width, label='Bat', align='center')

plt.xlabel('Seasons')
plt.ylabel('Noof')
plt.title('Toss Decision Across Seasons')
plt.xticks([i+ bar_width for i in x],seasons)
plt.legend()
plt.show()

23) Compare Total Played Matches vs Winning Matches vs Win Rate¶

In [83]:
Total_Matches_Counts = pd.concat([df['team1'],df['team2']]).value_counts()
Total_Matches = Total_Matches_Counts.index
Win_Matches_Count = df['winner'].value_counts()
data_group1=[] 
data_group2=[] 
teams=[]
for team in Total_Matches:
    teams.append(team)
    data_group1.append(Total_Matches_Counts.loc[team])
    data_group2.append(Win_Matches_Count.loc[team])

bar_width = 0.4
x = range(len(teams))

plt.bar(x, data_group1, width=bar_width, label='Total Played Matches', align='center')
plt.bar([i + bar_width for i in x], data_group2, width=bar_width, label='Winning Matches', align='center')

plt.xlabel('Teams')
plt.ylabel('Frequency')
plt.title('Total Played Matches vs Winning Matches')
plt.xticks([i for i in x], teams, rotation=45, ha='right')
plt.legend()
plt.show()
